 1 /******************************************************************************
 2  * Top contributors (to current version):
 3  *   Abdalrhman Mohamed, Mudathir Mohamed, Aina Niemetz
 4  *
 5  * This file is part of the cvc5 project.
 6  *

 9  * All rights reserved.  See the file COPYING in the top-level source
10  * directory for licensing information.
11  * ****************************************************************************
12  *
13  * A simple demonstration of the Sygus API.
14  *
15  * A simple demonstration of how to use the Sygus API to synthesize a simple
16  * invariant. Here is the same problem written in Sygus V2 format:
17  *
18  * (set-logic LIA)
19  *
20  * (synth-inv inv-f ((x Int)))
21  *
22  * (define-fun pre-f ((x Int)) Bool
23  *   (= x 0))
24  * (define-fun trans-f ((x Int) (xp Int)) Bool
25  *   (ite (< x 10) (= xp (+ x 1)) (= xp x)))
26  * (define-fun post-f ((x Int)) Bool
27  *   (<= x 10))
28  *
29  * (inv-constraint inv-f pre-f trans-f post-f)
30  *
31  * (check-synth)
32  *
33  * The printed output for this example should be equivalent to:
34  * (
35  *   (define-fun inv-f ((x Int)) Bool (not (>= x 11)))
36  * )
37  */
38 
39 #include <cvc5/cvc5.h>
40 
41 #include <iostream>
42 
43 #include "utils.h"
44 
45 using namespace cvc5::api;
46 
47 int main()
48 {
49   Solver slv;
50 
51   // required options
52   slv.setOption("lang", "sygus2");
53   slv.setOption("incremental", "false");
54 
55   // set the logic
56   slv.setLogic("LIA");
57 
58   Sort integer = slv.getIntegerSort();
59   Sort boolean = slv.getBooleanSort();
60 
61   Term zero = slv.mkInteger(0);
62   Term one = slv.mkInteger(1);
63   Term ten = slv.mkInteger(10);
64 
65   // declare input variables for functions
66   Term x = slv.mkVar(integer, "x");
67   Term xp = slv.mkVar(integer, "xp");
68 
69   // (ite (< x 10) (= xp (+ x 1)) (= xp x))
70   Term ite = slv.mkTerm(ITE,
71                         slv.mkTerm(LT, x, ten),
72                         slv.mkTerm(EQUAL, xp, slv.mkTerm(PLUS, x, one)),
73                         slv.mkTerm(EQUAL, xp, x));
74 
75   // define the pre-conditions, transition relations, and post-conditions
76   Term pre_f = slv.defineFun("pre-f", {x}, boolean, slv.mkTerm(EQUAL, x, zero));
77   Term trans_f = slv.defineFun("trans-f", {x, xp}, boolean, ite);
78   Term post_f = slv.defineFun("post-f", {x}, boolean, slv.mkTerm(LEQ, x, ten));
79 
80   // declare the invariant-to-synthesize
81   Term inv_f = slv.synthInv("inv-f", {x});
82 
83   slv.addSygusInvConstraint(inv_f, pre_f, trans_f, post_f);
84 
85   // print solutions if available
86   if (slv.checkSynth().isUnsat())
87   {
88     // Output should be equivalent to:
89     // (
90     //   (define-fun inv-f ((x Int)) Bool (not (>= x 11)))
91     // )
92     std::vector<Term> terms = {inv_f};
93     printSynthSolutions(terms, slv.getSynthSolutions(terms));
94   }
95 
96   return 0;
97 }
