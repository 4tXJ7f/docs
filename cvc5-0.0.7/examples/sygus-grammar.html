  2  * Top contributors (to current version):

 11  * ****************************************************************************
 12  *
 13  * A simple demonstration of the Sygus API.
 14  *
 15  * A simple demonstration of how to use Grammar to add syntax constraints to
 16  * the Sygus solution for the identity function. Here is the same problem
 17  * written in Sygus V2 format:
 18  *
 19  * (set-logic LIA)
 20  *
 21  * (synth-fun id1 ((x Int)) Int
 22  *   ((Start Int)) ((Start Int ((- x) (+ x Start)))))
 23  *
 24  * (synth-fun id2 ((x Int)) Int
 25  *   ((Start Int)) ((Start Int ((Variable Int) (- x) (+ x Start)))))
 26  *
 27  * (synth-fun id3 ((x Int)) Int
 28  *   ((Start Int)) ((Start Int (0 (- x) (+ x Start)))))
 29  *
 30  * (synth-fun id4 ((x Int)) Int
 31  *   ((Start Int)) ((Start Int ((- x) (+ x Start)))))
 32  *
 33  * (declare-var x Int)
 34  *
 35  * (constraint (= (id1 x) (id2 x) (id3 x) (id4 x) x))
 36  *
 37  * (check-synth)
 38  *
 39  * The printed output for this example should look like:
 40  * (
 41  *   (define-fun id1 ((x Int)) Int (+ x (+ x (- x))))
 42  *   (define-fun id2 ((x Int)) Int x)
 43  *   (define-fun id3 ((x Int)) Int (+ x 0))
 44  *   (define-fun id4 ((x Int)) Int (+ x (+ x (- x))))
 45  * )
 46  */
 47 
 48 #include <cvc5/cvc5.h>
 49 
 50 #include <iostream>
 51 
 52 #include "utils.h"
 53 
 54 using namespace cvc5::api;
 55 
 56 int main()
 57 {
 58   Solver slv;
 59 
 60   // required options
 61   slv.setOption("lang", "sygus2");
 62   slv.setOption("incremental", "false");
 63 
 64   // set the logic
 65   slv.setLogic("LIA");
 66 
 67   Sort integer = slv.getIntegerSort();
 68   Sort boolean = slv.getBooleanSort();
 69 
 70   // declare input variable for the function-to-synthesize
 71   Term x = slv.mkVar(integer, "x");
 72 
 73   // declare the grammar non-terminal
 74   Term start = slv.mkVar(integer, "Start");
 75 
 76   // define the rules
 77   Term zero = slv.mkInteger(0);
 78   Term neg_x = slv.mkTerm(UMINUS, x);
 79   Term plus = slv.mkTerm(PLUS, x, start);
 80 
 81   // create the grammar object
 82   Grammar g1 = slv.mkSygusGrammar({x}, {start});
 83 
 84   // bind each non-terminal to its rules
 85   g1.addRules(start, {neg_x, plus});
 86 
 87   // copy the first grammar with all of its non-termainals and their rules
 88   Grammar g2 = g1;
 89   Grammar g3 = g1;
 90 
 91   // add parameters as rules for the start symbol. Similar to "(Variable Int)"
 92   g2.addAnyVariable(start);
 93 
 94   // declare the functions-to-synthesize
 95   Term id1 = slv.synthFun("id1", {x}, integer, g1);
 96   Term id2 = slv.synthFun("id2", {x}, integer, g2);
 97 
 98   g3.addRule(start, zero);
 99 
100   Term id3 = slv.synthFun("id3", {x}, integer, g3);
101 
102   // g1 is reusable as long as it remains unmodified after first use
103   Term id4 = slv.synthFun("id4", {x}, integer, g1);
104 
105   // declare universal variables.
106   Term varX = slv.mkSygusVar(integer, "x");
107 
108   Term id1_x = slv.mkTerm(APPLY_UF, id1, varX);
109   Term id2_x = slv.mkTerm(APPLY_UF, id2, varX);
110   Term id3_x = slv.mkTerm(APPLY_UF, id3, varX);
111   Term id4_x = slv.mkTerm(APPLY_UF, id4, varX);
112 
113   // add semantic constraints
114   // (constraint (= (id1 x) (id2 x) (id3 x) (id4 x) x))
115   slv.addSygusConstraint(slv.mkTerm(EQUAL, {id1_x, id2_x, id3_x, id4_x, varX}));
116 
117   // print solutions if available
118   if (slv.checkSynth().isUnsat())
119   {
120     // Output should be equivalent to:
121     // (
122     //   (define-fun id1 ((x Int)) Int (+ x (+ x (- x))))
123     //   (define-fun id2 ((x Int)) Int x)
124     //   (define-fun id3 ((x Int)) Int (+ x 0))
125     //   (define-fun id4 ((x Int)) Int (+ x (+ x (- x))))
126     // )
127     std::vector<Term> terms = {id1, id2, id3, id4};
128     printSynthSolutions(terms, slv.getSynthSolutions(terms));
129   }
130 
131   return 0;
132 }
